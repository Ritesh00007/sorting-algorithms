Four Sorting Algorithms

Insertion Sort

Insertion sort is a simple sorting algorithm that builds the sorted array one item at a time. It has a time complexity of O(n²) → O(n) for best case. It is much like sorting playing cards in your hands. 

// Insertion Sort Implementation
function insertionSort(arr) {
    for (let i = 1; i < arr.length; i++) {
        let key = arr[i];
        let j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
    return arr;
}

Bubble Sort

Bubble sort is a simple sorting algorithm that repeatedly steps through the input list & compares adjacent elements. It swaps them if they are in the wrong order (∀ elements where a[i] > a[i+1]). Time complexity: O(n²). 

// Bubble Sort Implementation
function bubbleSort(arr) {
    let len = arr.length;
    for (let i = 0; i < len; i++) {
        for (let j = 0; j < len - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                let temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    return arr;
}

Selection Sort

Selection sort divides the input list into sorted & unsorted regions. It finds min(unsorted region) → moves it to the sorted region. Time complexity: Ω(n²) in all cases, making it inefficient for large lists. 

// Selection Sort Implementation
function selectionSort(arr) {
    let len = arr.length;
    for (let i = 0; i < len - 1; i++) {
        let minIdx = i;
        for (let j = i + 1; j < len; j++) {
            if (arr[j] < arr[minIdx]) {
                minIdx = j;
            }
        }
        if (minIdx !== i) {
            // Swap elements
            let temp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = temp;
        }
    }
    return arr;
}

Merge Sort

Merge sort is an efficient, stable algorithm using divide-and-conquer strategy. Process: array → split → sort → merge. Time complexity: O(n·log n) ∀ cases, making it more efficient than O(n²) algorithms. 

// Merge Sort Implementation
function mergeSort(arr) {
    if (arr.length <= 1) return arr;

    const mid = Math.floor(arr.length / 2);
    const left = arr.slice(0, mid);
    const right = arr.slice(mid);

    return merge(mergeSort(left), mergeSort(right));
}

function merge(left, right) {
    let result = [];
    let leftIndex = 0;
    let rightIndex = 0;

    while (leftIndex < left.length && rightIndex < right.length) {
        if (left[leftIndex] < right[rightIndex]) {
            result.push(left[leftIndex]);
            leftIndex++;
        } else {
            result.push(right[rightIndex]);
            rightIndex++;
        }
    }

    return result
        .concat(left.slice(leftIndex))
        .concat(right.slice(rightIndex));
}
